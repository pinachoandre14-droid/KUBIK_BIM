import React, { useEffect, useRef, useCallback } from "react"

export default function ScrollSequenceClaude() {
  const sectionRef = useRef<HTMLDivElement | null>(null)
  const iframeRef = useRef<HTMLIFrameElement | null>(null)

  const cloudflareUrl = "https://1841c7e8-floral-bird-adad.a01752509.workers.dev/"

  const sendProgress = useCallback(() => {
    const section = sectionRef.current
    const iframe = iframeRef.current

    if (!section || !iframe || !iframe.contentWindow) return

    const rect = section.getBoundingClientRect()
    const distance = section.offsetHeight - window.innerHeight

    const progress =
      distance <= 0 ? 0 : Math.min(Math.max(-rect.top / distance, 0), 1)

    iframe.contentWindow.postMessage(
      {
        type: "SCROLL_SEQUENCE_PROGRESS",
        progress,
      },
      "*"
    )
  }, [])

  useEffect(() => {
    let ticking = false

    const handleScroll = () => {
      if (!ticking) {
        window.requestAnimationFrame(() => {
          sendProgress()
          ticking = false
        })
        ticking = true
      }
    }

    window.addEventListener("scroll", handleScroll, { passive: true })
    window.addEventListener("resize", sendProgress)

    sendProgress()

    return () => {
      window.removeEventListener("scroll", handleScroll)
      window.removeEventListener("resize", sendProgress)
    }
  }, [sendProgress])

  return (
    <section
      ref={sectionRef}
      style={{
        position: "relative",
        width: "100vw",
        height: "1000vh",
        marginLeft: "calc(50% - 50vw)",
        marginRight: "calc(50% - 50vw)",
      }}
    >
      <div
        style={{
          position: "sticky",
          top: 0,
          width: "100vw",
          height: "100vh",
          overflow: "hidden",
        }}
      >
        <iframe
          ref={iframeRef}
          src={cloudflareUrl}
          onLoad={sendProgress}
          style={{
            width: "100%",
            height: "100%",
            border: "none",
            display: "block",
            pointerEvents: "none",
          }}
        />
      </div>
    </section>
  )
}
